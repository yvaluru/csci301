\sect{Recursion {\em vs.} Tail-recursion}

\[ a^b = \left\{\begin{array}{ll} 1 & \mbox{ if $b=0$}\\
    a(a^{b-1}) & \mbox{ otherwise}
  \end{array}\right.
  \]

\begin{Verbatim}
(define pow-rec
  (lambda (a b)
    (if (zero? b) 
        1
        (* a (pow-rec a (- b 1))))))
\end{Verbatim}
\vfill\pause

\begin{Verbatim}
(define pow-iter
  (lambda (a b)
    (define pow-iter-loop
      (lambda (b product)
        (if (zero? b)
            product
            (loop (- b 1) (* a product)))))
    (loop b 1)))
\end{Verbatim}
\end{frame}

\sect{Named {\tt let}}

\begin{Verbatim}
(define pow-iter
  (lambda (a b)
    (define loop
      (lambda (b product)
        (if (zero? b)
            product
            (loop (- b 1) (* a product)))))
    (loop b 1)))
\end{Verbatim}
\vfill
\begin{Verbatim}
(define pow-iter-2
  (lambda (a b)
    (let loop ((b b) (product 1))
      (if (zero? b)
          product
          (loop (- b 1) (* a product))))))
\end{Verbatim}
\end{frame}

\sect{Fast recursion}
\begin{Verbatim}
(define pow-fast
  (lambda (a b)
    (cond ((zero? b) 1)
          ((even? b) (sqr (pow-fast a (/ b 2))))
          (else (* a (pow-fast a (- b 1)))))))
\end{Verbatim}
\end{frame}

\sect{nth}
\pause
\begin{Verbatim}
(define (nth lst n)
  (cond ((null? lst) nil)
        ((= n 0) "Not defined")
        ((= n 1) (car lst))
        (else (nth (cdr lst) (- n 1)))))
\end{Verbatim}
\end{frame}
\sect{last}
\pause
\begin{Verbatim}
(define (last lst)
  (cond ((null? lst) nil)
        ((null? (cdr lst)) (car lst))
        (else (last (cdr lst)))))
\end{Verbatim}
\end{frame}
\sect{scale-list}
\pause
\begin{Verbatim}
(define (scale-list lst n)
  (if (null? lst)
      nil
      (cons (* n (car lst))
            (scale-list (cdr lst) n))))
\end{Verbatim}
\end{frame}
\sect{increment-list}
\pause
\begin{Verbatim}

      (cons (+ 1 (car lst))

\end{Verbatim}
\end{frame}
\sect{map}
\pause
\begin{Verbatim}
(define (map lst op)
  (if (null? lst)
      nil
      (cons (op (car lst))
            (map (cdr lst) op))))
\end{Verbatim}
\end{frame}

\sect{append}
\pause
\begin{Verbatim}
(define (append lst1 lst2)
  (if (null? lst1)
      lst2
      (cons (car lst1)
            (append (cdr lst1) lst2))))
\end{Verbatim}
\end{frame}
\sect{remove}
\pause
\begin{Verbatim}
(define (remove n lst)
  (cond ((null? lst) nil)
        ((= n (car lst)) (remove n (cdr lst)))
        (else (cons (car lst)
                    (remove n (cdr lst))))))
\end{Verbatim}
\end{frame}

\sect{count-leaves}
\pause
\begin{Verbatim}
(define (count-leaves tree)
  (cond ((null? tree) 0)
        ((not (pair? tree)) 1)
        (else (+ (count-leaves (car tree))
                 (count-leaves (cdr tree))))))
\end{Verbatim}

\sect{fringe}
\pause
\begin{Verbatim}
(define (fringe tree)
  (cond ((null? tree) nil)
        ((not (pair? tree)) (list tree))
        (else (append (fringe (car tree))
                      (fringe (cdr tree))))))
\end{Verbatim}
\end{frame}

(define (sum-fringe tree)

\end{frame}
\sect{map-tree}

(define (map-tree tree op)

        (else (cons (map-tree (car tree) op)
                    (map-tree (cdr tree) op)))))
\end{Verbatim}
\end{frame}
